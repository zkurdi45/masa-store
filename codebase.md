from django.db import models
from django.contrib.auth.models import AbstractUser
from django.utils.translation import gettext_lazy as _
# settings.AUTH_USER_MODEL is not typically imported into models.py directly
# from django.conf import settings # Not needed here for CustomUser definition

class CustomUser(AbstractUser):
    # AbstractUser already provides:
    # username, first_name, last_name, email, password, groups, user_permissions,
    # is_staff, is_active, is_superuser, last_login, date_joined.

    # We want to use email as the primary identifier and make it unique.
    email = models.EmailField(_('email address'), unique=True)
    
    # 'name' field for the user's full name, as 'first_name' and 'last_name' might be too granular
    # or you might prefer a single field.
    name = models.CharField(_('Full name'), max_length=255, blank=True, help_text=_('Full name of the user.'))

    ROLE_CHOICES = [
        ('staff', 'Staff'),
        ('customer', 'Customer'),
    ]
    role = models.CharField(
        max_length=10,
        choices=ROLE_CHOICES,
        default='customer', # It's good practice to have a default
        help_text=_('User role in the system.')
    )

    USERNAME_FIELD = 'email' # Use email to log in
    REQUIRED_FIELDS = ['username', 'name'] # Fields prompted for when creating a user via createsuperuser
                                        # 'username' is from AbstractUser. Even if email is the USERNAME_FIELD,
                                        # 'username' can still be required by Django if not made optional on the model.
                                        # If you truly want to get rid of 'username', AbstractBaseUser is the way.

    def __str__(self):
        return self.email # Or self.name, depending on preference

    # You can add other custom methods or properties here

# Other models (Product, Basket, BasketItem) remain the same
class Product(models.Model):
    name = models.CharField(max_length=255)
    price = models.DecimalField(max_digits=10, decimal_places=2)

    def __str__(self):
        return self.name

class Basket(models.Model):
    STATUS_CHOICES = [
        ('active', 'Active'),
        ('submitted', 'Submitted'),
        ('abandoned', 'Abandoned'),
    ]
    user = models.ForeignKey(CustomUser, on_delete=models.CASCADE, related_name='baskets')
    status = models.CharField(max_length=10, choices=STATUS_CHOICES, default='active')

    def __str__(self):
        return f"Basket {self.id} for {self.user} ({self.status})"

class BasketItem(models.Model):
    basket = models.ForeignKey(Basket, on_delete=models.CASCADE, related_name='items')
    product = models.ForeignKey(Product, on_delete=models.CASCADE)
    quantity = models.PositiveIntegerField(default=1)

    class Meta:
        unique_together = ('basket', 'product')

    def __str__(self):
        return f"{self.quantity} of {self.product.name} in Basket {self.basket.id}"

from django.http import HttpResponse
from django.urls import reverse_lazy # For generating URLs in the welcome message

# This view is now in its own file.
def home_view(request):
    welcome_message = "Welcome to the Grocery Store!"
    user_display_name = "Guest" # Default for anonymous users
    if request.user.is_authenticated:
        # Try to get 'name' attribute, fallback to 'username'
        user_display_name = getattr(request.user, 'name', None) or request.user.username
        welcome_message += f" Hello, {user_display_name}!"

    html_content = f"<h1>{welcome_message}</h1>"
    # Generate URLs safely using reverse_lazy
    if request.user.is_authenticated:
        try:
            logout_url = reverse_lazy('store:logout')
            html_content += f"<p><a href=\"{logout_url}\">Logout</a></p>"
        except Exception:
            html_content += "<p>Logout link not available</p>" # Fallback
    else:
        try:
            login_url = reverse_lazy('store:login')
            register_url = reverse_lazy('store:register')
            html_content += f"<p><a href=\"{login_url}\">Login</a> | <a href=\"{register_url}\">Register</a></p>"
        except Exception:
            html_content += "<p>Login/Register links not available</p>" # Fallback
            
    return HttpResponse(html_content)
